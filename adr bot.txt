    function addMsg(msg, type) {
        let div = document.createElement("div");
        div.className = "msg " + type;
        div.innerHTML = msg + `<div class="time">${new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })}</div>`;
        chat.appendChild(div);
        chat.scrollTop = chat.scrollHeight;
    }

    window.onload = () => {
        addMsg("<b>🛡️ Welcome to Suraksha Vigil Bot</b><br><br>Smart ADR Reporting Assistant.<br><br>कृपया भाषा निवडा / Please choose language.", "bot");
        showLanguageButtons();
    };

    function showLanguageButtons() {
        let wrap = document.createElement("div");
        wrap.className = "chips";
        let b1 = document.createElement("button"); b1.className = "chip"; b1.innerText = "🇮🇳 मराठी"; b1.onclick = () => setLang("mr");
        let b2 = document.createElement("button"); b2.className = "chip"; b2.innerText = "🇬🇧 English"; b2.onclick = () => setLang("en");
        wrap.appendChild(b1); wrap.appendChild(b2);
        chat.appendChild(wrap);
    }

    function setLang(l) {
        lang = l;
        addMsg(l === "mr" ? "मराठी निवडली" : "English Selected", "user");
        chatStarted = true; step = 0; data = {};
        setTimeout(askQuestion, 700);
    }

    function askQuestion() {
        let qs = lang === "mr" ? mrQuestions : enQuestions;
        if (step < qs.length) {
            addMsg(qs[step].q, "bot");
            if (qs[step].chips) showChips(qs[step].chips);
        } else {
            addMsg(lang === "mr" ? "✅ तुमचा विस्तृत रिपोर्ट जमा झाला आहे." : "✅ Your detailed report has been submitted.", "bot");
            sendToSheet(data);
        }
    }

    function showChips(arr) {
        let wrap = document.createElement("div"); wrap.className = "chips";
        arr.forEach(x => {
            let b = document.createElement("button"); b.className = "chip"; b.innerText = x;
            b.onclick = () => handleInput(x);
            wrap.appendChild(b);
        });
        chat.appendChild(wrap);
    }

    function handleInput(v) {
        addMsg(v, "user");
        data["Q" + step] = v;
        step++;
        setTimeout(askQuestion, 500);
    }

    function send() {
        let val = document.getElementById("input").value.trim();
        if (!val || !chatStarted) return;
        handleInput(val);
        document.getElementById("input").value = "";
    }

    function sendToSheet(finalData) {
        // तुमचा नवीन Deployment URL इथे टाकला आहे
        const scriptURL = "https://script.google.com/macros/s/AKfycbxCjmGGBkMcVoolkSV88_eURfRtjqKn-A0nie_hYdXu-WGwCS3ARIa4vSU3vvf752AKiQ/exec";
        fetch(scriptURL, {
            method: "POST",
            mode: "no-cors",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify(finalData)
        }).then(() => console.log("Success")).catch(e => console.log(e));
    }

    // Call / Voice Logic (Original)
    function startCall() { document.getElementById("callScreen").style.display = "flex"; }
    function endCall() { document.getElementById("callScreen").style.display = "none"; }
    function voiceInput() {
        let recog = new webkitSpeechRecognition();
        recog.lang = lang === "en" ? "en-IN" : "mr-IN";
        recog.start();
        recog.onresult = (e) => handleInput(e.results[0][0].transcript);
    }
